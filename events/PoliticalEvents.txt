# Class Conflicts Divide [Root.GetName]
country_event = {
	id = political.4
	title = political.4.t
	desc = political.4.d
	picture = GFX_report_event_gathering_protest
	
	trigger = {
		has_idea_with_trait = hive_mind__revolutionary
		NOT = { has_government = hive_mind_ }
		hive_mind_ > 0.2
		NOT = { hive_mind_ > 0.59 }
		if = {
			limit = {
				has_dlc = "Together for Victory"
			}
			OR = {
				NOT = { tag = RAJ }
				AND = {
					tag = RAJ
					is_puppet = no
				}
			}
		}
		if = {
			limit = {
				has_dlc = "Man the Guns"
			}
			not = { tag = USA }
		}
		NOT = {
			AND = {
				has_dlc = "No Step Back"
				OR = {
					original_tag = POL
					original_tag = EST
					original_tag = LAT
					original_tag = LIT
					AND = {
						original_tag = FIN
						EST = { has_completed_focus = EST_national_pride_in_finland }
					}
				}
			}
		}
		NOT = {
			AND = {
				has_dlc = "Battle for the Bosporus"
				OR = {
					original_tag = GRE
					original_tag = TUR
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.5
			has_country_flag = hive_mind__popular_movement
		}
	}
	
	option = {
		name = political.4.a
		add_popularity = {
			ideology = hive_mind_
			popularity = 0.05
		}
		add_stability = -0.05
	}
}
